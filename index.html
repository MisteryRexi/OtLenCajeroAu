            // Código en Java
    
            import java.util.Scanner;
    
            public class CajeroAutomatico {
                int moneda10 =0;
                int moneda5 =0;
                int moneda2 =0;
                int moneda1 =0;
                int costo =0;
                int opcion =0;
                int pago =0;
                int cambio =0;
                int horas =0;
                String variable = null;
    
                public void run(){
                    Scanner scanner =new Scanner(System.in);
    
                    while(true){
                        costo =0;
                        horas =0;
                        System.out.println("Estacionamiento Internacional Los Ángeles");
                        System.out.println("¿Cuántas horas se utilizo el estacionamiento?");
                        horas = scanner.nextInt();
                        costo = horas*17;
    
                        if(costo > 999){
                            costo = 999;
                        }
                        pago =0;
                        opcion =0;
                        while(!(pago >= costo)){
                            System.out.println("Debes: $" + (costo - pago));
                            System.out.println("Elije una opción para tu forma de pago:");
                            System.out.println("1)$100 | 2)$50 | 3)$20 | 4)$10 | 5)$5 | 6)$2 | 7)$1");
                            opcion = scanner.nextInt();
                            if(opcion == 1){
                                pago = pago + 100;
                            }
                            if(opcion == 2){
                                pago = pago + 50;
                            }
                            if(opcion == 3){
                                pago = pago + 20;
                            }
                            if(opcion == 4){
                                pago = pago + 10;
                            }
                            if(opcion == 5){
                                pago = pago + 5;
                            }
                            if(opcion == 6){
                                pago = pago + 2;
                            }
                            if(opcion == 7){
                                pago = pago + 1;
                            }
                            cambio =0;
                            moneda10 =0;
                            moneda5 =0;
                            moneda2= 0;
                            moneda1 =0;
                            cambio = pago - costo;
                            if(cambio > 0){
                                moneda10 = (cambio / 10);
                                cambio = cambio % 10;
                            }
                            if(cambio > 0){
                                moneda5 = (cambio / 5);
                                cambio = cambio % 5;
                            }
                            if(cambio > 0){
                                moneda2 = (cambio / 2);
                                cambio = cambio % 2;
                            }
                            if(cambio > 0){
                                moneda1 = (cambio / 1);
                                cambio = cambio % 1;
                            }
                        }
                        System.out.println("Cambio a recibir: ");
                        System.out.println(moneda10 + " moneda(s) de 10");
                        System.out.println(moneda5 + " moneda(s) de 5");
                        System.out.println(moneda2 + " moneda(s) de 2");
                        System.out.println(moneda1 + " moneda(s) de 1");
                        System.out.println("El cambio total es: " + cambio);
                    }
                }
    
                public static void main(String[] args) throws Exception {
                    CajeroAutomatico cajeroautomatico = new CajeroAutomatico();
